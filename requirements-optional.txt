# Optional dependencies for psws-drf-tid-tools
# These improve specific outputs but aren't strictly required.

# Better-looking maps with state/country outlines (tid_map.py).
# Without cartopy, tid_map.py falls back to a plain lat/lon plot.
cartopy>=0.21

# Required for cwt-prophet extraction method — the recommended extraction
# method used by tid_spect_click.py (Pass 0 auto-run) and drf_to_doppler.py
# --method cwt-prophet. Without prophet, cwt-prophet falls back to cwt-only.
# Note: prophet has its own dependencies (pystan, cmdstanpy) that may
# require separate installation steps on some platforms.
# See: https://facebook.github.io/prophet/docs/installation.html
prophet>=1.1

# Used by hf_int.py's Lomb-Scargle periodogram (Step 2, TID period
# detection) for a proper false-alarm-probability significance test.
# Without astropy, hf_int.py still runs, falling back to scipy's
# signal.lombscargle with an approximate power-threshold heuristic
# instead of a real statistical confidence check.
astropy>=5.0

# Required by fetch_madrigal_tec.py (GPS TEC cross-correlation for
# independent DOA corroboration) — hard requirement for that script
# specifically, not a soft fallback; it exits immediately if missing.
# Not needed for the core extraction/DOA pipeline.
madrigalWeb>=3.2
